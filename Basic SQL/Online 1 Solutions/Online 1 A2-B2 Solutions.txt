SELECT D1.DEPARTMENT_ID, D1.DEPARTMENT_NAME, (L1.STREET_ADDRESS || ', ' || L1.CITY || ', ' || STATE_PROVINCE || ', ' || POSTAL_CODE) ADDRESS
FROM DEPARTMENTS D1 
JOIN LOCATIONS L1 ON D1.LOCATION_ID = L1.LOCATION_ID 
ORDER BY D1.DEPARTMENT_ID

2. 

SELECT TO_CHAR(HIRE_DATE,'Day') Day,count(*) Total_Employees
From EMPLOYEES
GROUP BY TO_CHAR(HIRE_DATE,'Day'),TO_CHAR(HIRE_DATE,'D')
ORDER BY MOD(TO_CHAR(HIRE_DATE,'D')+1, 7)


3. 

SELECT E1.EMPLOYEE_ID, (E1.FIRST_NAME || ' ' || E1.LAST_NAME) NAME, 
E1.SALARY, COUNT(DISTINCT E2.EMPLOYEE_ID) "RICHER THAN", 
COUNT(DISTINCT E3.EMPLOYEE_ID) "POORER THAN" 
FROM EMPLOYEES E1 
    LEFT OUTER JOIN EMPLOYEES E2 ON E1.SALARY > E2.SALARY 
    LEFT OUTER JOIN EMPLOYEES E3 ON E1.SALARY < E3.SALARY
GROUP BY E1.EMPLOYEE_ID, E1.SALARY, E1.FIRST_NAME, E1.LAST_NAME
HAVING COUNT(DISTINCT E2.EMPLOYEE_ID) > 10 and COUNT(DISTINCT E3.EMPLOYEE_ID) <= 25
ORDER BY E1.EMPLOYEE_ID

SELECT FIRST_NAME, LPAD(SALARY, 10,'$') SALARY,DEPARTMENT_ID,Replace(DEPARTMENT_NAME,' ','') DEPARTMENT_NAME
FROM employees E JOIN DEPARTMENTS D Using (DEPARTMENT_ID)
WHERE DEPARTMENT_ID BETWEEN 10 and 90
ORDER BY LENGTH(DEPARTMENT_NAME) DESC

SELECT SUBSTR(FIRST_NAME, 1, 1) || SUBSTR(LAST_NAME, 1, 1) || ' ' || EMPLOYEE_ID EMPLOYEE_code, MONTHS_BETWEEN(TRUNC(SYSDATE, 'Month'),TRUNC(HIRE_DATE, 'Month')) * SALARY Earned_SALARY
FROM EMPLOYEES
WHERE (SYSDATE-HIRE_DATE) > (365*18)
ORDER BY EMPLOYEE_ID Desc
