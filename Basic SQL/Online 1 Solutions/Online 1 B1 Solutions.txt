Select (FIRST_NAME || ' ' || LAST_NAME) FullName, SALARY
From EMPLOYEES
WHERE LENGTH(FIRST_NAME || LAST_NAME) > 10
ORDER BY SALARY

2. 

SELECT TO_CHAR(HIRE_DATE, 'MONTH') Month_Name,Count(*) Total_Employees
FROM EMPLOYEES
GROUP BY TO_CHAR(HIRE_DATE, 'MONTH'),TO_CHAR(HIRE_DATE,'mm')
ORDER BY TO_CHAR(HIRE_DATE,'mm')

3. 

SELECT E1.MANAGER_ID MANAGER_ID, Count(*) as Total_Employee, Round(AVG(E1.SALARY),2) Avg_Sal
FROM EMPLOYEES E1 JOIN EMPLOYEES E2
ON (E1.MANAGER_ID = E2.EMPLOYEE_ID)
GROUP BY E1.MANAGER_ID
HAVING Count(*) > 4 and AVG(E1.SALARY) > 3000
ORDER BY Total_Employee Desc, Avg_Sal Asc

4.

SELECT D.DEPARTMENT_NAME, count(*) Total_Employees, Round(avg(E.SALARY),2) Avg_Sal
From DEPARTMENTS D Join EMPLOYEES E 
on E.DEPARTMENT_ID = D.DEPARTMENT_ID
GROUP BY D.DEPARTMENT_NAME
ORDER BY Total_Employees Desc

SELECT (SUBSTR(FIRST_NAME, 1,1) || '. '  || LAST_NAME) FullName, (SALARY*12 + (SALARY*12) * NVL(COMMISSION_PCT,0)) Yearly_Income
FROM EMPLOYEES
WHERE LAST_NAME Like 'm%' or LAST_NAME LIKE 'M%'
ORDER BY Yearly_Income Desc
