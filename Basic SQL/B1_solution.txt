1.
SELECT FIRST_NAME||' '||LAST_NAME FULLNAME, SALARY
FROM EMPLOYEES
WHERE LENGTH(FIRST_NAME||LAST_NAME)>10
ORDER BY SALARY ASC;

2.
SELECT TO_CHAR(HIRE_DATE, 'MONTH') MONTH_NAME, COUNT(EMPLOYEE_ID) TOTAL_EMPLOYEES
FROM EMPLOYEES
GROUP BY TO_CHAR(HIRE_DATE, 'MONTH')
ORDER BY TO_DATE(TO_CHAR(HIRE_DATE, 'MONTH'), 'MONTH');

3.
SELECT MANAGER_ID, COUNT(EMPLOYEE_ID) TOTAL_EMPLOYEES, ROUND(AVG(SALARY),2) AVG_SAL
FROM EMPLOYEES
GROUP BY MANAGER_ID
HAVING AVG(SALARY)>3000 AND MANAGER_ID IS NOT NULL AND COUNT(EMPLOYEE_ID)>4
ORDER BY COUNT(EMPLOYEE_ID) DESC, AVG(SALARY) ASC;
4.
SELECT D.DEPARTMENT_NAME, COUNT(E.EMPLOYEE_ID) TOTAL_EMPLOYEES, ROUND(AVG(E.SALARY), 2) AVG_SAL
FROM EMPLOYEES E JOIN DEPARTMENTS D
ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
GROUP BY D.DEPARTMENT_ID, D.DEPARTMENT_NAME
ORDER BY TOTAL_EMPLOYEES DESC;

5.
SELECT SUBSTR(FIRST_NAME, 1, 1)||'. '||LAST_NAME FULLNAME, SALARY*12+SALARY*12*NVL(COMMISSION_PCT, 0) YEARLY_INCOME
FROM EMPLOYEES
WHERE UPPER(LAST_NAME) LIKE 'M%'
ORDER BY YEARLY_INCOME DESC;
